import urllib2
import json

zabbixurl = 'http://192.168.2.91/zabbix/api_jsonrpc.php'
username = "Admin"
password = "zzjr#2015"



'''登陆函数获取登录auth'''
def authenticate(zabbixurl, username, password):
    values = {'jsonrpc': '2.0',
            'method': 'user.login',
            'params': {
                'user': username,
                'password': password
            },
            'id': '0'
            }

    data = json.dumps(values)
    req = urllib2.Request(zabbixurl, data, {'Content-Type': 'application/json-rpc'})
    response = urllib2.urlopen(req, data)
    output = json.loads(response.read())


    try:
        message = output['result']

    except:
        message = output['error']['data']
        quit()

    return output['result']

auth=authenticate(zabbixurl, username, password)
print auth

'''根据主机名获取hostid'''
def gethost():
    values={
    "jsonrpc": "2.0",
    "method": "host.get",
    "params": {
        "output": "extend",
        "filter": {
            "host": [
                "Zabbix server"
            ]
        }
    },
    "auth": auth,
    "id": 1
}
    data = json.dumps(values)
    req = urllib2.Request(zabbixurl, data, {'Content-Type': 'application/json-rpc'})
    response = urllib2.urlopen(req, data)
    host_get = response.read()
    host=json.loads(host_get)
    return host

hostid=gethost()['result'][0]['hostid']
print hostid
